        function showToast(message, type = 'info', duration = 2200) {
            const el = document.createElement('div');
            el.className = `fixed bottom-6 right-6 px-5 py-3 rounded-2xl border text-sm flex items-center gap-3 z-[300] shadow-xl ${type === 'success' ? 'bg-emerald-600 border-emerald-500 text-white' : 'bg-slate-700 border-slate-600 text-slate-200'}`;
            el.innerHTML = `<i class="fa-solid ${type === 'success' ? 'fa-check-circle' : 'fa-info-circle'} mr-2"></i> ${message}`;
            document.body.appendChild(el);
            setTimeout(() => { el.style.transition = 'all 0.2s'; el.style.opacity = '0'; setTimeout(() => el.remove(), 200); }, duration);
        }
