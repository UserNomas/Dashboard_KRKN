    <script>
        function switchPlatformTab(tabElement, platform) {
            document.querySelectorAll('.platform-tab').forEach(t => t.classList.remove('active', 'border-emerald-500', 'text-white'));
            document.querySelectorAll('.platform-tab').forEach(t => t.classList.add('text-slate-300'));
            
            tabElement.classList.add('active', 'border-emerald-500', 'text-white');
            tabElement.classList.remove('text-slate-300');
            
            // For demo, only Instagram content is fully built. Show toast for others.
            if (platform !== 'instagram') {
                showToast(`Switching to ${platform.charAt(0).toUpperCase() + platform.slice(1)} view...`, 'info');
            }
        }
        
        function startInstagramScan() {
            const username = document.getElementById('target-username').value || 'jperez_ig';
            showToast(`Starting deep scan on @${username}...`, 'info');
            
            setTimeout(() => {
                showToast(`Scan completed • 342 posts • 47 media items extracted`, 'success');
            }, 2100);
        }
        
        function startNewScan() {
            showToast('Opening new multi-platform scan wizard...', 'info');
        }
        
        function downloadAllMedia() {
            showToast('Preparing archive (2.4 GB) • Download started', 'success');
        }
        
        function showToast(message, type = 'info', duration = 2200) {
            const el = document.createElement('div');
            el.className = `fixed bottom-6 right-6 px-5 py-3 rounded-2xl border text-sm flex items-center gap-3 z-[300] shadow-xl ${type === 'success' ? 'bg-emerald-600 border-emerald-500 text-white' : 'bg-slate-700 border-slate-600 text-slate-200'}`;
            el.innerHTML = `<i class="fa-solid ${type === 'success' ? 'fa-check-circle' : 'fa-info-circle'} mr-2"></i> ${message}`;
            document.body.appendChild(el);
            setTimeout(() => { el.style.transition = 'all 0.2s'; el.style.opacity = '0'; setTimeout(() => el.remove(), 200); }, duration);
        }
        
        function initialize() {
            console.log('%c[SOCINT] OSINT Hub initialized', 'color:#64748b');
        }
        window.onload = initialize;
